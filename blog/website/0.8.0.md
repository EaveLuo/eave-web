function extractDescription(frontMatterDesc, content, maxLength = 120) {
  // 优先级 1: front matter description
  if (frontMatterDesc?.trim()) {
    return truncate(frontMatterDesc, maxLength);

  // 优先级 2: 从正文提取（去除 markdown 标记）
  const text = content
    .replace(/```[\s\S]*?```/g, ' ')     // 代码块

    .replace(/!\[[^\]]*\]\([^)]+\)/g, ' ') // 图片
    .replace(/\[([^\]]+)\]\([^)]+\)/g, '$1') // 链接

}

### ⚡ 动画性能优化

#### 核心问题诊断

通过性能分析发现首页存在以下瓶颈：

| 组件 | 问题 | 影响 |
|------|------|------|
| Particles | 100 粒子 + 每帧重绘 + 高频鼠标事件 | GPU 占用高，低端设备卡顿 |
| TypingAnimation | setInterval 16ms 强制重排 | 主线程阻塞 |
| ArticleCard | 12 张卡片同时触发 Framer Motion | 大量合成层，内存占用高 |
| Name 鼠标追踪 | 每次 mousemove 都 setProperty | 强制样式重计算 |

#### 优化策略与实现

**1. Particles 粒子效果**

```typescript
// 优化前
quantity: 100        // 粒子数量
animate: setInterval // 定时器驱动

// 优化后
quantity: 30         // 减少 70%
animate: requestAnimationFrame // 帧同步

// 新增优化
const isActive = useRef(true);           // 生命周期控制
const dpr = Math.min(window.devicePixelRatio, 2); // 限制 DPR
useCallback + useMemo                    // 缓存计算
RAF 节流鼠标事件                          // 降低事件频率
```

**2. TypingAnimation 打字效果**

```typescript
// 优化前
useEffect(() => {
  const interval = setInterval(() => {
    setDisplayedText(prev => prev + text.charAt(i));
    setI(i + 1);
  }, duration);
}, [i]);

// 优化后
const animate = useCallback((currentTime: number) => {
  const deltaTime = currentTime - lastTimeRef.current;
  if (deltaTime >= duration) {
    setDisplayedText(text.slice(0, indexRef.current + 1));
    indexRef.current += 1;
    lastTimeRef.current = currentTime;
  }
  rafRef.current = requestAnimationFrame(animate);
}, []);
```

**3. ArticleCard 卡片入场动画**

```typescript
// 优化前 - Framer Motion whileInView
<motion.article
  initial={{ opacity: 0, y: 20 }}
  whileInView={{ opacity: 1, y: 0 }}
  viewport={{ once: true }}
  transition={{ duration: 0.4, delay: index * 0.1 }}
/>

// 优化后 - CSS 动画 + useInView
const isInView = useInView(cardRef, { once: true, margin: "-50px" });

// CSS
.card {
  opacity: 0;
  transform: translateY(20px);
  transition: opacity 0.4s ease, transform 0.4s ease;
}
.cardVisible {
  opacity: 1;
  transform: translateY(0);
}
```

**4. Hero 鼠标光效**

```typescript
// 优化前
onMouseMove={(e) => {
  e.currentTarget.style.setProperty('--x', `${e.clientX}px`);
  e.currentTarget.style.setProperty('--y', `${e.clientY}px`);
}}

// 优化后
const handleMouseMove = useCallback((e: React.MouseEvent) => {
  if (rafRef.current) return;  // RAF 节流
  rafRef.current = requestAnimationFrame(() => {
    setMousePos({ x: e.clientX - bounding.x, y: e.clientY - bounding.y });
    rafRef.current = 0;
  });
}, []);
```

#### 关键代码片段

**延迟加载粒子效果：**

```typescript
// 使用 IntersectionObserver + requestIdleCallback
useEffect(() => {
  const observer = new IntersectionObserver((entries) => {
    entries.forEach((entry) => {
      if (entry.isIntersecting) {
        requestIdleCallback(() => setShowParticles(true), { timeout: 2000 });
        observer.disconnect();
      }
    });
  });
  
  // 3 秒后备方案
  const fallback = setTimeout(() => setShowParticles(true), 3000);
}, []);
```

### 🎉 总结

0.8.0 版本是一次**体验与性能并重**的升级。首页卡片模块让内容展示更加直观，文档规范化确保了多语言用户的无缝切换，而深度性能优化则让整体体验更加丝滑流畅。

特别值得一提的是，所有优化都遵循了**渐进增强**原则：
- 支持 `prefers-reduced-motion` 媒体查询，尊重用户偏好
- 使用 `requestIdleCallback` 延迟非关键资源加载
- 保留原始组件作为 fallback，便于回滚

性能优化没有终点，只有持续迭代 🚀
