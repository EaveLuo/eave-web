# Concurrency Patterns - Practice中的Best Practices

并发模式是解决常见并发问题的标准方案。掌握这些模式，让你的并发代码更可靠。

## 🎯 Basics模式

### 1. 生成器模式

```go
// 生成数字序列
func generate(n int) <-chan int {
    ch := make(chan int)
    
    go func() {
        for i := 0; i < n; i++ {
            ch <- i
        }
        close(ch)
    }()
    
    return ch
}

// 使用
for num := range generate(10) {
    fmt.Println(num)
}
```

### 2. 管道模式

```go
// 阶段 1：生成
func generate(nums []int) <-chan int {
    out := make(chan int)
    go func() {
        for _, n := range nums {
            out <- n
        }
        close(out)
    }()
    return out
}

// 阶段 2：处理
func square(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        for n := range in {
            out <- n * n
        }
        close(out)
    }()
    return out
}

// 阶段 3：收集
func collect(in <-chan int) []int {
    var result []int
    for n := range in {
        result = append(result, n)
    }
    return result
}

// 组合
nums := []int{1, 2, 3, 4, 5}
ch := generate(nums)
ch = square(ch)
result := collect(ch)
```

### 3. 扇出扇入模式

```go
// 扇出：分发任务
func fanOut(in <-chan int, n int) []<-chan int {
    outs := make([]<-chan int, n)
    for i := 0; i < n; i++ {
        out := make(chan int)
        outs[i] = out
        
        go func(o chan<- int) {
            for v := range in {
                o <- v
            }
            close(o)
        }(out)
    }
    return outs
}

// 扇入：汇聚结果
func fanIn(overs ...<-chan int) <-chan int {
    out := make(chan int)
    var wg sync.WaitGroup
    
    for _, ch := range overs {
        wg.Add(1)
        go func(c <-chan int) {
            defer wg.Done()
            for v := range c {
                out <- v
            }
        }(ch)
    }
    
    go func() {
        wg.Wait()
        close(out)
    }()
    
    return out
}
```

## 🔧 错误处理模式

### 1. 错误收集

```go
type Result struct {
    value int
    err   error
}

func worker(id int, jobs <-chan int) <-chan Result {
    out := make(chan Result)
    
    go func() {
        for job := range jobs {
            // 可能出错的工作
            if job < 0 {
                out <- Result{err: fmt.Errorf("无效任务：%d", job)}
            } else {
                out <- Result{value: job * 2}
            }
        }
        close(out)
    }()
    
    return out
}
```

### 2. 错误传播

```go
func processAll(jobs []int) error {
    jobCh := make(chan int, len(jobs))
    resultCh := make(chan Result, len(jobs))
    
    // 启动 Worker
    for i := 0; i < 3; i++ {
        go func() {
            for result := range worker(i, jobCh) {
                resultCh <- result
            }
        }()
    }
    
    // 发送任务
    for _, job := range jobs {
        jobCh <- job
    }
    close(jobCh)
    
    // 收集错误
    var errors []error
    for i := 0; i < len(jobs); i++ {
        result := <-resultCh
        if result.err != nil {
            errors = append(errors, result.err)
        }
    }
    
    if len(errors) > 0 {
        return fmt.Errorf("发生%d个错误", len(errors))
    }
    return nil
}
```

## 🚦 限流模式

### 1. 信号量限流

```go
type Semaphore chan struct{}

func NewSemaphore(n int) Semaphore {
    return make(Semaphore, n)
}

func (s Semaphore) Acquire() {
    s <- struct{}{}
}

func (s Semaphore) Release() {
    <-s
}

// 使用
sem := NewSemaphore(10)

for i := 0; i < 100; i++ {
    go func(id int) {
        sem.Acquire()
        defer sem.Release()
        
        // 执行任务（最多 10 个并发）
        process(id)
    }(i)
}
```

### 2. 令牌桶限流

```go
type TokenBucket struct {
    tokens   int
    capacity int
    rate     time.Duration
    timer    *time.Ticker
}

func NewTokenBucket(capacity int, rate time.Duration) *TokenBucket {
    tb := &TokenBucket{
        tokens:   capacity,
        capacity: capacity,
        rate:     rate,
        timer:    time.NewTicker(rate),
    }
    
    go func() {
        for range tb.timer.C {
            if tb.tokens < tb.capacity {
                tb.tokens++
            }
        }
    }()
    
    return tb
}

func (tb *TokenBucket) Take() {
    <-tb.timer.C
}
```

## 🛑 Cancel模式

### 1. Context Cancel

```go
func worker(ctx context.Context, id int) {
    for {
        select {
        case <-ctx.Done():
            fmt.Printf("Worker %d 被取消\n", id)
            return
        default:
            // 工作
            time.Sleep(100 * time.Millisecond)
        }
    }
}

// 使用
ctx, cancel := context.WithCancel(context.Background())

for i := 0; i < 5; i++ {
    go worker(ctx, i)
}

time.Sleep(time.Second)
cancel()  // 取消所有 Worker
```

### 2. Timeout控制

```go
func processWithTimeout(ctx context.Context, timeout time.Duration) error {
    ctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    
    done := make(chan error)
    
    go func() {
        // 耗时操作
        time.Sleep(2 * time.Second)
        done <- nil
    }()
    
    select {
    case err := <-done:
        return err
    case <-ctx.Done():
        return fmt.Errorf("超时：%v", timeout)
    }
}
```

## 🎨 Practice模式

### 1. 重试模式

```go
func retry(ctx context.Context, maxRetries int, fn func() error) error {
    var lastErr error
    
    for i := 0; i < maxRetries; i++ {
        select {
        case <-ctx.Done():
            return ctx.Err()
        default:
        }
        
        if err := fn(); err != nil {
            lastErr = err
            time.Sleep(time.Duration(i+1) * time.Second)  // 指数退避
            continue
        }
        return nil
    }
    
    return lastErr
}

// 使用
err := retry(context.Background(), 3, func() error {
    return connectToDatabase()
})
```

### 2. 批量处理

```go
func batchProcess(items []int, batchSize int, processFn func(int) error) error {
    for i := 0; i < len(items); i += batchSize {
        end := i + batchSize
        if end > len(items) {
            end = len(items)
        }
        
        batch := items[i:end]
        var wg sync.WaitGroup
        errCh := make(chan error, len(batch))
        
        for _, item := range batch {
            wg.Add(1)
            go func(n int) {
                defer wg.Done()
                if err := processFn(n); err != nil {
                    errCh <- err
                }
            }(item)
        }
        
        wg.Wait()
        close(errCh)
        
        // 检查错误
        if err := <-errCh; err != nil {
            return err
        }
    }
    
    return nil
}
```

### 3. 优先级队列

```go
type PriorityTask struct {
    priority int
    task     func()
}

func priorityWorker(highPriority, lowPriority <-chan PriorityTask) {
    for {
        select {
        case task := <-highPriority:
            task.task()
        case task := <-lowPriority:
            task.task()
        }
    }
}
```

## 💡 Best Practices

### 1. 始终Close Channel

```go
func producer(ch chan<- int) {
    defer close(ch)  // 确保关闭
    
    for i := 0; i < 10; i++ {
        ch <- i
    }
}
```

### 2. 使用 Context

```go
// ✅ 推荐
func process(ctx context.Context) error {
    // ...
}

// ❌ 避免
func process2() error {
    // 无法取消
}
```

### 3. 避免 Goroutine 泄漏

```go
// ✅ 正确
func doWork() error {
    done := make(chan struct{})
    
    go func() {
        // 工作
        close(done)
    }()
    
    select {
    case <-done:
        return nil
    case <-time.After(time.Minute):
        return errors.New("超时")
    }
}
```
