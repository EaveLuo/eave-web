# Concurrency Patterns - Best Practices in Practice

Concurrency patterns are standard solutions to common concurrency problems. Master these patterns to make your concurrent code more reliable.

## Target Basic Patterns

### 1. Generator Pattern

```go
// Generate number sequence
func generate(n int) <-chan int {
    ch := make(chan int)
    
    go func() {
        for i := 0; i < n; i++ {
            ch <- i
        }
        close(ch)
    }()
    
    return ch
}

// Usage
for num := range generate(10) {
    fmt.Println(num)
}
```

### 2. Pipeline Pattern

```go
// Stage 1: Generate
func generate(nums []int) <-chan int {
    out := make(chan int)
    go func() {
        for _, n := range nums {
            out <- n
        }
        close(out)
    }()
    return out
}

// Stage 2: Process
func square(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        for n := range in {
            out <- n * n
        }
        close(out)
    }()
    return out
}

// Stage 3: Collect
func collect(in <-chan int) []int {
    var result []int
    for n := range in {
        result = append(result, n)
    }
    return result
}

// Compose
nums := []int{1, 2, 3, 4, 5}
ch := generate(nums)
ch = square(ch)
result := collect(ch)
```

### 3. Fan-out Fan-in Pattern

```go
// Fan-out: Distribute tasks
func fanOut(in <-chan int, n int) []<-chan int {
    outs := make([]<-chan int, n)
    for i := 0; i < n; i++ {
        out := make(chan int)
        outs[i] = out
        
        go func(o chan<- int) {
            for v := range in {
                o <- v
            }
            close(o)
        }(out)
    }
    return outs
}

// Fan-in: Aggregate results
func fanIn(overs ...<-chan int) <-chan int {
    out := make(chan int)
    var wg sync.WaitGroup
    
    for _, ch := range overs {
        wg.Add(1)
        go func(c <-chan int) {
            defer wg.Done()
            for v := range c {
                out <- v
            }
        }(ch)
    }
    
    go func() {
        wg.Wait()
        close(out)
    }()
    
    return out
}
```

## HammerAndWrench Error Handling Patterns

### 1. Error Collection

```go
type Result struct {
    value int
    err   error
}

func worker(id int, jobs <-chan int) <-chan Result {
    out := make(chan Result)
    
    go func() {
        for job := range jobs {
            // Work that may fail
            if job < 0 {
                out <- Result{err: fmt.Errorf("invalid task: %d", job)}
            } else {
                out <- Result{value: job * 2}
            }
        }
        close(out)
    }()
    
    return out
}
```

### 2. Error Propagation

```go
func processAll(jobs []int) error {
    jobCh := make(chan int, len(jobs))
    resultCh := make(chan Result, len(jobs))
    
    // Start Workers
    for i := 0; i < 3; i++ {
        go func() {
            for result := range worker(i, jobCh) {
                resultCh <- result
            }
        }()
    }
    
    // Send tasks
    for _, job := range jobs {
        jobCh <- job
    }
    close(jobCh)
    
    // Collect errors
    var errors []error
    for i := 0; i < len(jobs); i++ {
        result := <-resultCh
        if result.err != nil {
            errors = append(errors, result.err)
        }
    }
    
    if len(errors) > 0 {
        return fmt.Errorf("%d errors occurred", len(errors))
    }
    return nil
}
```

## VerticalTrafficLight Rate Limiting Patterns

### 1. Semaphore Rate Limiting

```go
type Semaphore chan struct{}

func NewSemaphore(n int) Semaphore {
    return make(Semaphore, n)
}

func (s Semaphore) Acquire() {
    s <- struct{}{}
}

func (s Semaphore) Release() {
    <-s
}

// Usage
sem := NewSemaphore(10)

for i := 0; i < 100; i++ {
    go func(id int) {
        sem.Acquire()
        defer sem.Release()
        
        // Execute task (max 10 concurrent)
        process(id)
    }(i)
}
```

### 2. Token Bucket Rate Limiting

```go
type TokenBucket struct {
    tokens   int
    capacity int
    rate     time.Duration
    timer    *time.Ticker
}

func NewTokenBucket(capacity int, rate time.Duration) *TokenBucket {
    tb := &TokenBucket{
        tokens:   capacity,
        capacity: capacity,
        rate:     rate,
        timer:    time.NewTicker(rate),
    }
    
    go func() {
        for range tb.timer.C {
            if tb.tokens < tb.capacity {
                tb.tokens++
            }
        }
    }()
    
    return tb
}

func (tb *TokenBucket) Take() {
    <-tb.timer.C
}
```

## StopSign Cancellation Patterns

### 1. Context Cancellation

```go
func worker(ctx context.Context, id int) {
    for {
        select {
        case <-ctx.Done():
            fmt.Printf("Worker %d cancelled\n", id)
            return
        default:
            // Work
            time.Sleep(100 * time.Millisecond)
        }
    }
}

// Usage
ctx, cancel := context.WithCancel(context.Background())

for i := 0; i < 5; i++ {
    go worker(ctx, i)
}

time.Sleep(time.Second)
cancel()  // Cancel all Workers
```

### 2. Timeout Control

```go
func processWithTimeout(ctx context.Context, timeout time.Duration) error {
    ctx, cancel := context.WithTimeout(ctx, timeout)
    defer cancel()
    
    done := make(chan error)
    
    go func() {
        // Time-consuming operation
        time.Sleep(2 * time.Second)
        done <- nil
    }()
    
    select {
    case err := <-done:
        return err
    case <-ctx.Done():
        return fmt.Errorf("timeout: %v", timeout)
    }
}
```

## ArtistPalette Practical Patterns

### 1. Retry Pattern

```go
func retry(ctx context.Context, maxRetries int, fn func() error) error {
    var lastErr error
    
    for i := 0; i < maxRetries; i++ {
        select {
        case <-ctx.Done():
            return ctx.Err()
        default:
        }
        
        if err := fn(); err != nil {
            lastErr = err
            time.Sleep(time.Duration(i+1) * time.Second)  // Exponential backoff
            continue
        }
        return nil
    }
    
    return lastErr
}

// Usage
err := retry(context.Background(), 3, func() error {
    return connectToDatabase()
})
```

### 2. Batch Processing

```go
func batchProcess(items []int, batchSize int, processFn func(int) error) error {
    for i := 0; i < len(items); i += batchSize {
        end := i + batchSize
        if end > len(items) {
            end = len(items)
        }
        
        batch := items[i:end]
        var wg sync.WaitGroup
        errCh := make(chan error, len(batch))
        
        for _, item := range batch {
            wg.Add(1)
            go func(n int) {
                defer wg.Done()
                if err := processFn(n); err != nil {
                    errCh <- err
                }
            }(item)
        }
        
        wg.Wait()
        close(errCh)
        
        // Check errors
        if err := <-errCh; err != nil {
            return err
        }
    }
    
    return nil
}
```

### 3. Priority Queue

```go
type PriorityTask struct {
    priority int
    task     func()
}

func priorityWorker(highPriority, lowPriority <-chan PriorityTask) {
    for {
        select {
        case task := <-highPriority:
            task.task()
        case task := <-lowPriority:
            task.task()
        }
    }
}
```

## LightBulb Best Practices

### 1. Always Close Channels

```go
func producer(ch chan<- int) {
    defer close(ch)  // Ensure close
    
    for i := 0; i < 10; i++ {
        ch <- i
    }
}
```

### 2. Use Context

```go
// WhiteCheckMark Recommended
func process(ctx context.Context) error {
    // ...
}

// CrossMark Avoid
func process2() error {
    // Cannot cancel
}
```

### 3. Avoid Goroutine Leaks

```go
// WhiteCheckMark Correct
func doWork() error {
    done := make(chan struct{})
    
    go func() {
        // Work
        close(done)
    }()
    
    select {
    case <-done:
        return nil
    case <-time.After(time.Minute):
        return errors.New("timeout")
    }
}
```
